<script>
    $('span.pin').on('click', function () {
        if ($(this).attr("data-pinned") == 1) {
            var message = 'Убрать это сообщение с главной страницы договора? Cообщение останется в логах по клиенту.';
            if (!confirm(message)) {
                return;
            }
            $.post('/work2/ajax/unpin/' + $(this).attr("data-id"), {}, function (json) {
                if (json.result != 'ok') {
                    alert("Не удалось открепить сообщение");
                } else {
                    window.location.reload();
                }
            }, 'json');
        } else {
            var message = 'Прикрепить сообщение?';
            if (!confirm(message)) {
                return;
            }
            $.post('/work2/ajax/pin/' + $(this).attr("data-id"), {}, function (json) {
                if (json.result != 'ok') {
                    alert("Не удалось прикрепить сообщение");
                } else {
                    window.location.reload();
                }
            }, 'json');
        }
    });
</script>
